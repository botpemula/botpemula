<?php

class botpemula
{
    private $useragent;
    private $cookie;
    
    public function __construct()
    {
        $this->useragent = 'Mozilla/5.0 (X11; Ubuntu; Linux i686; rv:40.0)';
        $this->cookie = '';
    }

    private function requestHeaders($method = 'GET')
    {
        $headers = [
            'user-agent: ' . $this->useragent,
            'cookie: ' . $this->cookie,
        ];
                      
        if ($method == 'POST')
        {
            $headers[] = 'content-type: application/x-www-form-urlencoded';
        }
                      
        return $headers;
    }
}

?>
